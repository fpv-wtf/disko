00184 
00185                     // normals

00187                                          sintheta * cosphi,
00188                                          sinphi,
00189                                          costheta * cosphi);

00191                     // vertices
00192                     MMS_VA_SET_VERTEX_3v(vertices, index, x, y, z);

00195                 }
00196

00200         }
